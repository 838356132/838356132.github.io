let cv = new Canvas('cv', [0,360], [640,0]);

function loadImage(src, callback) {
    let image = document.createElement('img');
    image.src = src;
    image.addEventListener("load", function() {
        cv.context.drawImage(image, 0, 0);
        callback();
    });
}
